\section{Contoh Kode}
\begin{verbatim}
; Program demonstrasi prosedur dan makro
TITLE Pemrograman Modular
.MODEL SMALL
.STACK 100h

.DATA
    pesan1 DB 'Hello from procedure!$'
    pesan2 DB 'Hello from macro!$'
    nilai1 DW 15
    nilai2 DW 25
    hasil DW ?

; Makro untuk menampilkan pesan
tampilkan_pesan MACRO pesan
    PUSH AX
    PUSH DX
    MOV AH, 09h
    MOV DX, OFFSET pesan
    INT 21h
    POP DX
    POP AX
ENDM

; Makro untuk pertukaran nilai
tukar_nilai MACRO var1, var2
    PUSH AX
    MOV AX, var1
    XCHG AX, var2
    MOV var1, AX
    POP AX
ENDM

.CODE
START:
    MOV AX, @DATA
    MOV DS, AX
    
    ; Panggil prosedur
    CALL tampilkan_hello
    
    ; Gunakan makro
    tampilkan_pesan pesan2
    
    ; Pertukaran nilai menggunakan makro
    tukar_nilai nilai1, nilai2
    
    ; Panggil prosedur dengan parameter
    PUSH nilai1
    PUSH nilai2
    CALL tambahkan
    ADD SP, 4
    
    MOV AH, 4Ch
    INT 21h

; Prosedur tanpa parameter
tampilkan_hello PROC
    PUSH AX
    PUSH DX
    
    MOV AH, 09h
    MOV DX, OFFSET pesan1
    INT 21h
    
    POP DX
    POP AX
    RET
tampilkan_hello ENDP

; Prosedur dengan parameter
tambahkan PROC
    PUSH BP
    MOV BP, SP
    
    PUSH AX
    PUSH BX
    
    MOV AX, [BP+6]  ; Parameter kedua
    MOV BX, [BP+4]  ; Parameter pertama
    ADD AX, BX
    MOV hasil, AX
    
    POP BX
    POP AX
    POP BP
    RET
tambahkan ENDP

END START
\end{verbatim}
